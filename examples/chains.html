    for attempt in range(max_retries):
        try:
            result = chain.run(question)
            return result
        except Exception as e:
            print(f"–ü–æ–ø—ã—Ç–∫–∞ {attempt + 1} –Ω–µ —É–¥–∞–ª–∞—Å—å: {e}")

                raise e
